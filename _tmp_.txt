docker run -p 2012:3306 --name=mysql-db-taskmgr -e MYSQL_ROOT_PASSWORD=secret1 -e MYSQL_USER=test -e MYSQL_PASSWORD=test123 -e MYSQL_DATABASE=taskmanager -d mysql

docker exec -it mysql-db-taskmgr bash

mysql -utest -ptest123
